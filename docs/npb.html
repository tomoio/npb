module.exports = class NPB
  constructor: (@root = process.cwd(), @opts = {}) ->
    @basename = path.basename(@root)
    @projectTitle = @opts.projectitle ? @basename
    @description = @opts.description ? 'A node module.'

    fs.mkdir(@root, (err) =>
      @mkdir()
    )

    @GITIGNORE = ['lib-cov', '*.seed', '*.log', '*.csv', '*.dat', '*.out', '*.pid', '*.gz', '', 'pids', 'logs', 'results', '', 'node_modules', 'npm-debug.log']

    @MAKEFILE = [
      'test:',
      '  ./node_modules/.bin/mocha ./test/*.coffee --compilers coffee:coffee-script -R spec',
      '.PHONY: test'
    ]

    @PACKAGE_JSON = {
      "name" : @basename,
      "version" : "0.0.1",
      "description" : @description,
      "keywords" : [],
      "bugs" : {"url" : "https://github.com/tomoio/#{@basename}/issues"},
      "author" : {
        "name" : "Tomo I/O",
        "email" : "711@tomo.io",
        "url" : "http://tomo.io"
      },
      "main" : "./lib/#{@basename}",
      "directories" : {
        "lib" : "./lib",
        "doc" : "./docs"
      },
      "repository": {
        "type": "git",
        "url": "https://github.com/tomoio/#{@basename}"
      },
      "dependencies" : {
      },
      "devDependencies" : {
        "coffee-script" : "",
        "mocha" : "",
        "chai" : ""
      }
    }

    if @opts.bin
      @PACKAGE_JSON.bin = "./bin/#{@basename}"
      @PACKAGE_JSON.directories.bin = "./bin"
      @PACKAGE_JSON.dependencies.commander = "*"

    @BYSTANDER = {
      "ignore" : ["**/node_modules","**/.", "**/assets", "**/test/tmp"],   
      "plugins" : ["by-coffeescript", "by-write2js", "by-coffeelint", "by-docco", "by-mocha"],
      "by" : {
        "coffeescript" : {"noCompile" : ["**/test/*"]
        },
        "write2js" : {
          "bin" : true,
          "mapper" : {
            "**/src/*" : ["/src/", "/lib/"]
          }
        },
        "docco" : {
          "doccoSources" : ["src/*.coffee"]
        },
        "mocha" : {
          "testPaths" : ["test/*.coffee"]
        }
      }
    }

  mkdir : ->
    dirs = ['src', 'docs', 'test', 'lib']
    extra = ['bin','assets']
    for v in extra
      if @opts[v]?
        dirs.push(v)
    async.forEach(
      dirs,
      (v, callback) =>
        fs.mkdir(path.join(@root,v), (err) =>
          console.log(err)
          callback()
        )
        
      (cb) =>
        @writeGitignore()
    )

  writeGitignore : ->
    fs.writeFile(path.join(@root,'.gitignore'),@GITIGNORE.join('\n'), (err) =>
      console.log(err)
      console.log('fuckkkk')
      @writeMakefile()
    )

  writeMakefile : ->
    fs.writeFile(path.join(@root,'Makefile'),@MAKEFILE.join('\n'), (err) =>
      console.log('mkfile')
      console.log(err)
      @writePackageJson()
    )

  writePackageJson : ->
    fs.writeFile(path.join(@root,'package.json'),beautifyjs.js_beautify(JSON.stringify(@PACKAGE_JSON),{indent_size:2}), (err) =>
      console.log('mkfile')
      console.log(err)
      @writeBystander()
    )

  writeBystander : ->
    fs.writeFile(path.join(@root,'.bystander'),beautifyjs.js_beautify(JSON.stringify(@BYSTANDER),{indent_size:2}), (err) =>
      console.log('mkfile')
      console.log(err)
    )
